### 保存镜像
docker image save nginx:1.11 > nginx1.11.tar

du -sh nginx1.11.tar

docker load < nginx1.11.tar


### 导出容器 
docker ps

docker run -itd busybox

docker export sha1Id busybox.tar  


### 导入容器
docker image import busybox.tar busybox:self
